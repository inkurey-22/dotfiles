cmake_minimum_required(VERSION 3.10)
project(amazed)

set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 17)

# Add source files
file(GLOB_RECURSE SOURCES
    "src/*.c"
    "src/*.cpp"
)

# Add header files (optional, for IDEs)
file(GLOB_RECURSE HEADERS
    "include/*.h"
    "include/*.hpp"
)

# Add lib sources
file(GLOB_RECURSE LIB_SOURCES
    "lib/*.c"
    "lib/*.cpp"
)

# Create static library libmy.a
add_library(my STATIC ${LIB_SOURCES})
set_target_properties(my PROPERTIES OUTPUT_NAME "my")

# Add executable
add_executable(amazed ${SOURCES} ${HEADERS})

# Include directories
target_include_directories(amazed PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}/include
    ${CMAKE_CURRENT_SOURCE_DIR}/lib
)

# Link static library to executable
target_link_libraries(amazed PRIVATE my)
